    public enum State {
    NEW,           // мғқм„ұлҗЁ, м•„м§Ғ мӢңмһ‘ м•ҲлҗЁ
    RUNNABLE,      // мӢӨн–ү к°ҖлҠҘ мғҒнғң (CPU н• лӢ№ лҢҖкё° лҳҗлҠ” мӢӨн–ү мӨ‘)
    BLOCKED,       // synchronized лқҪ лҢҖкё°
    WAITING,       // лӢӨлҘё мҠӨл Ҳл“ңмқҳ мӢ нҳё л¬ҙкё°н•ң лҢҖкё°
    TIMED_WAITING, // лӢӨлҘё мҠӨл Ҳл“ңмқҳ мӢ нҳё лҳҗлҠ” нғҖмһ„м•„мӣғ лҢҖкё°
    TERMINATED     // мӢӨн–ү мҷ„лЈҢ
}

System.out.println(thread.getState());  // RUNNABLE

    public class Main {
    public static void main(String[] args) throws InterruptedException {
        Thread worker = new Thread(() -> {
            try {
                System.out.println("мһ‘м—… мӢңмһ‘");

                System.out.println("мһ‘м—… мҷ„лЈҢ");

    wait() / notify(): мҠӨл Ҳл“ң к°„ мӢ нҳё

    public class Buffer {
    private Queue<String> queue = new LinkedList<>();
    private int capacity = 5;

    // м•„мқҙн…ң м¶”к°Җ
    public synchronized void put(String item) throws InterruptedException {
        while (queue.size() >= capacity) {
            wait();  // лІ„нҚј к°Җл“қ м°Ёл©ҙ лҢҖкё°
        }
        queue.add(item);
        System.out.println("м¶”к°Җ: " + item);
        notifyAll();  // лҢҖкё° мӨ‘мқё мҠӨл Ҳл“ң к№Ёмҡ°кё°
    }

    // м•„мқҙн…ң м ңкұ°
    public synchronized String get() throws InterruptedException {
        while (queue.isEmpty()) {
            wait();  // лІ„нҚј л№„л©ҙ лҢҖкё°
        }
        String item = queue.poll();
        System.out.println("м ңкұ°: " + item);
        notifyAll();  // лҢҖкё° мӨ‘мқё мҠӨл Ҳл“ң к№Ёмҡ°кё°
        return item;
    }
}

// мӮ¬мҡ©
Buffer buffer = new Buffer();

// Producer (м¶”к°Җн•ҳлҠ” мҠӨл Ҳл“ң)
new Thread(() -> {
    for (int i = 0; i < 10; i++) {
        try {
            buffer.put("м•„мқҙн…ң " + i);
            Thread.sleep(100);
        } catch (InterruptedException e) {}

}).start();

    }
}).start();

// лӘЁл“  мӣҢм»Ө мҷ„лЈҢ лҢҖкё°
for (Thread worker : workers) {
    worker.join();
}
System.out.println("лӘЁл“  мһ‘м—… мҷ„лЈҢ");

    4. wait() vs sleep() vs yield()

    л©”м„ңл“ң лӘ©м Ғ лқҪ к№Ёмҡ°кё°
    sleep() мӢңк°„ кІҪкіј лҢҖкё° мң м§Җ нғҖмһ„м•„мӣғ
    wait() мЎ°кұҙ лҢҖкё° н•ҙм ң notify()
    yield() CPU м–‘ліҙ мң м§Җ мҰүмӢң 
    synchronized void example() {
    Thread.sleep(1000);    // лқҪ мң м§Җн•ҳл©ҙм„ң 1мҙҲ лҢҖкё°

    while (condition == false) {
        wait();            // лқҪ н•ҙм ңн•ҳкі  лҢҖкё°, notify мӢң лӢӨмӢң нҡҚл“қ
    }

    Thread.yield();        // лқҪ мң м§Җн•ҳл©ҙм„ң лӢӨлҘё мҠӨл Ҳл“ңм—җ м–‘ліҙ
}

    мЈјмқҳн•  м җ

    вқҢ notify()мҷҖ notifyAll() нҳјлҸҷ

    synchronized void wrongNotification() {
    notify();  // вқҢ лҢҖкё° мӨ‘мқё м—¬лҹ¬ мҠӨл Ҳл“ң мӨ‘ 1к°ңл§Ң к№ЁмӣҖ
    // к№Ём–ҙлӮң мҠӨл Ҳл“ңк°Җ мЎ°кұҙмқ„ л§ҢмЎұн•ҳм§Җ м•Ҡмңјл©ҙ, лӢӨлҘё мҠӨл Ҳл“ңлҠ” мҳҒкө¬ лҢҖкё°
}

synchronized void correctNotification() {
    notifyAll();  // вң… лӘЁл“  лҢҖкё° мҠӨл Ҳл“ң к№Ёмҡ°кё°
    // к°Ғ мҠӨл Ҳл“ңк°Җ лӢӨмӢң мЎ°кұҙ нҷ•мқё (while лЈЁн”„)
}

    вқҢ wait()лҘј ifлЎң мӮ¬мҡ©

    synchronized void wrong() {
    if (queue.isEmpty()) {
        wait();  // вқҢ к№Ём–ҙлӮҳл©ҙ л°”лЎң кі„мҶҚ (мЎ°кұҙ лҜёнҷ•мқё)
    }
    String item = queue.poll();  // м—¬кё°м„ң NPE л°ңмғқ к°ҖлҠҘ
}

synchronized void correct() {
    while (queue.isEmpty()) {
        wait();  // вң… к№Ём–ҙлӮң нӣ„ лӢӨмӢң мЎ°кұҙ нҷ•мқё
    }
    String item = queue.poll();  // м•Ҳм „
}

    // вқҢ sleep мӨ‘ лқҪ м җмң лЎң лӢӨлҘё мҠӨл Ҳл“ң лё”лЎңнӮ№

        // н•„мҡ”н•ң мһ‘м—…л§Ң
    }

    }
}

    рҹ’Ў мҠӨл Ҳл“ң м•Ҳм „н•ң лҢҖкё° нҢЁн„ҙ

    synchronized void waitForCondition() {
    while (!conditionMet) {  // н•ӯмғҒ while мӮ¬мҡ©
        try {
            wait();
        } catch (InterruptedException e) {
            Thread.currentThread().interrupt();  // мқён„°лҹҪнҠё мғҒнғң ліөмӣҗ
            break;
        }
    }
}

synchronized void signalCondition() {
    conditionMet = true;
    notifyAll();  // лӘЁл“  лҢҖкё° мҠӨл Ҳл“ң к№Ёмҡ°кё°
}
