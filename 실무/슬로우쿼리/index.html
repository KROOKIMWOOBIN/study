    슬로우 쿼리 (Slow Query)

    왜 공부하는가?

    서비스 규모가 커지면 처음에는 빠르던 쿼리가 데이터 증가와 함께 급격히 느려진다. 슬로우 쿼리는 API 응답 시간 저하, DB 서버 과부하, 전체 서비스 장애로 이어질 수 있다. 진단 → 원인 파악 → 최적화의 흐름을 체계적으로 알아야 한다.

    슬로우 쿼리란?

    실행 시간이 설정된 임계값을 초과하는 SQL 쿼리.

    일반적으로 MySQL에서 long_query_time (기본값 10초)을 초과하면 slow_query_log에 기록된다. 실무에서는 1초 이상 걸리는 쿼리부터 문제로 보고, 웹 API 기준으로는 100ms 이상도 최적화 대상이다.

    슬로우 쿼리 발생 원인

    1. 인덱스 미사용 (Full Table Scan)

    -- 인덱스 없는 컬럼으로 조회 → 테이블 전체 스캔
SELECT * FROM orders WHERE customer_name = '홍길동';
-- customer_name에 인덱스가 없으면 1천만 건 전부 탐색

-- 해결: 인덱스 추가
CREATE INDEX idx_orders_customer_name ON orders(customer_name);

    2. 잘못된 인덱스 사용 (Index 무효화)

    -- 함수를 사용하면 인덱스 무효화
SELECT * FROM users WHERE YEAR(created_at) = 2026;  -- BAD
SELECT * FROM users WHERE created_at >= '2026-01-01' AND created_at < '2027-01-01';  -- GOOD

-- 묵시적 형변환으로 인덱스 무효화
SELECT * FROM users WHERE user_id = '123';  -- user_id가 INT인데 문자열 비교 → BAD
SELECT * FROM users WHERE user_id = 123;    -- GOOD

-- 부정 조건은 인덱스 효율 저하
SELECT * FROM orders WHERE status != 'CANCELLED';  -- 주의
SELECT * FROM orders WHERE status IN ('PENDING', 'PAID', 'SHIPPED');  -- 더 명시적

    3. 대량 데이터 처리

    -- 대량 삭제/수정: 한 번에 처리하면 락 시간 길어짐
DELETE FROM logs WHERE created_at < '2025-01-01';  -- BAD (수백만 건 한 번에)

-- 배치로 나눠서 처리
DELETE FROM logs WHERE created_at < '2025-01-01' LIMIT 1000;  -- GOOD (반복 실행)

    4. 과도한 조인 (JOIN 과다)

    -- 조인이 많을수록 카테시안 곱 위험, 인덱스 효과 감소
SELECT u.*, o.*, p.*, r.*, d.*
FROM users u
JOIN orders o ON u.id = o.user_id
JOIN payments p ON o.id = p.order_id
JOIN reviews r ON o.id = r.order_id
JOIN deliveries d ON o.id = d.order_id
WHERE u.id = 1;
-- 조인마다 인덱스를 활용하는지 EXPLAIN으로 확인 필수

    5. N+1 쿼리 문제

    // JPA N+1: 1번의 쿼리로 N개의 엔티티를 가져온 후, 각 엔티티의 연관관계를 N번 추가 조회
List<Order> orders = orderRepository.findAll(); // 쿼리 1번
for (Order order : orders) {
    // 각 order의 user를 Lazy 로딩 → N번 추가 쿼리
    System.out.println(order.getUser().getName());
}
// 주문 100건이면 총 101번 쿼리 실행!

    -- 실제로 실행되는 쿼리
SELECT * FROM orders;                     -- 1번
SELECT * FROM users WHERE id = 1;        -- N번
SELECT * FROM users WHERE id = 2;
SELECT * FROM users WHERE id = 3;
...

    6. 불필요한 컬럼 조회 (SELECT *)

    -- BAD: 불필요한 컬럼까지 모두 조회 (네트워크, 메모리 낭비)
SELECT * FROM products WHERE category_id = 1;

-- GOOD: 필요한 컬럼만 조회
SELECT id, name, price FROM products WHERE category_id = 1;

    7. LIKE %검색어% (앞에 % 사용)

    -- BAD: 앞에 %를 사용하면 인덱스 사용 불가 (B-Tree 특성)
SELECT * FROM products WHERE name LIKE '%노트북%';

-- GOOD: 접두사 검색은 인덱스 활용 가능
SELECT * FROM products WHERE name LIKE '삼성%';

-- 전문 검색이 필요하면 Full-Text Index 사용
ALTER TABLE products ADD FULLTEXT INDEX idx_name_fulltext(name);
SELECT * FROM products WHERE MATCH(name) AGAINST('노트북' IN BOOLEAN MODE);

    진단 방법

    MySQL slow_query_log 설정

    -- 슬로우 쿼리 로그 활성화 (MySQL)
SET GLOBAL slow_query_log = 'ON';
SET GLOBAL long_query_time = 1;          -- 1초 이상 걸리는 쿼리 기록
SET GLOBAL slow_query_log_file = '/var/log/mysql/slow.log';
SET GLOBAL log_queries_not_using_indexes = 'ON';  -- 인덱스 미사용 쿼리도 기록

-- 현재 설정 확인
SHOW VARIABLES LIKE 'slow_query%';
SHOW VARIABLES LIKE 'long_query_time';

    # my.cnf에 영구 설정
[mysqld]
slow_query_log = 1
slow_query_log_file = /var/log/mysql/slow.log
long_query_time = 1
log_queries_not_using_indexes = 1

    # mysqldumpslow로 슬로우 쿼리 분석
mysqldumpslow -s t -t 10 /var/log/mysql/slow.log  # 실행 시간순 상위 10개
mysqldumpslow -s c -t 10 /var/log/mysql/slow.log  # 실행 횟수순 상위 10개

    EXPLAIN / EXPLAIN ANALYZE

    -- 실행 계획 확인
EXPLAIN SELECT * FROM orders o
JOIN users u ON o.user_id = u.id
WHERE o.status = 'PENDING'
AND o.created_at >= '2026-01-01';

    EXPLAIN 결과 주요 컬럼 설명

    +----+-------------+-------+------+---------------+------+---------+------+------+-------------+
| id | select_type | table | type | possible_keys | key  | key_len | ref  | rows | Extra       |
+----+-------------+-------+------+---------------+------+---------+------+------+-------------+
|  1 | SIMPLE      | o     | ALL  | NULL          | NULL | NULL    | NULL |10000 | Using where |
|  1 | SIMPLE      | u     | ref  | PRIMARY       | PRIMARY | 8    | o.user_id| 1 |         |
+----+-------------+-------+------+---------------+------+---------+------+------+-------------+

    type 컬럼 해석 (성능 좋은 순)

    type 의미 성능
    system 테이블에 행이 1개 최상
    const PK/Unique로 1건 조회 최상
    eq_ref JOIN 시 PK/Unique로 1건 매우 좋음
    ref 인덱스로 여러 건 조회 좋음
    range 인덱스 범위 스캔 보통
    index 인덱스 전체 스캔 나쁨
    ALL 테이블 전체 스캔 (Full Scan) 최악 
    -- EXPLAIN ANALYZE: 실제 실행 통계 포함 (MySQL 8.0+)
EXPLAIN ANALYZE
SELECT * FROM orders WHERE user_id = 1 AND status = 'PAID';

-- 결과 예시
-- -> Index lookup on orders using idx_user_status (user_id=1, status='PAID')
--    (cost=2.50 rows=5) (actual time=0.021..0.025 rows=3 loops=1)

    EXPLAIN 결과 해석 실습

    -- 예시 테이블
CREATE TABLE orders (
    id BIGINT PRIMARY KEY,
    user_id BIGINT NOT NULL,
    status VARCHAR(20) NOT NULL,
    total_amount DECIMAL(10,2),
    created_at DATETIME NOT NULL,
    INDEX idx_user_id (user_id),
    INDEX idx_status (status),
    INDEX idx_created_at (created_at)
);

-- 케이스 1: Full Table Scan (나쁨)
EXPLAIN SELECT * FROM orders WHERE total_amount > 100000;
-- type: ALL (total_amount에 인덱스 없음)

-- 케이스 2: 인덱스 범위 스캔 (보통)
EXPLAIN SELECT * FROM orders WHERE created_at >= '2026-01-01';
-- type: range (idx_created_at 사용)

-- 케이스 3: 복합 인덱스 활용 (좋음)
ALTER TABLE orders ADD INDEX idx_user_status (user_id, status);
EXPLAIN SELECT * FROM orders WHERE user_id = 1 AND status = 'PAID';
-- type: ref (idx_user_status 사용)

-- 케이스 4: 커버링 인덱스 (매우 좋음)
EXPLAIN SELECT id, user_id, status FROM orders WHERE user_id = 1;
-- type: ref, Extra: Using index (테이블 접근 없이 인덱스만으로 해결)

    최적화 방법

    1. 적절한 인덱스 추가

    -- 단일 컬럼 인덱스
CREATE INDEX idx_orders_status ON orders(status);

-- 복합 인덱스 (컬럼 순서가 중요! 카디널리티 높은 것 먼저)
CREATE INDEX idx_orders_user_status ON orders(user_id, status);
-- user_id로 먼저 필터링, 그 중에서 status로 필터링

-- 복합 인덱스 활용 규칙: 왼쪽 접두사 규칙
-- idx_orders_user_status(user_id, status)가 있을 때:
SELECT * FROM orders WHERE user_id = 1;              -- 인덱스 사용 O (user_id만 조건)
SELECT * FROM orders WHERE user_id = 1 AND status = 'PAID';  -- 인덱스 사용 O (둘 다 조건)
SELECT * FROM orders WHERE status = 'PAID';          -- 인덱스 사용 X (user_id가 앞에 있어야 함)

    2. N+1 쿼리 해결

    // 문제 코드: N+1 발생
@Entity
public class Order {
    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "user_id")
    private User user;
}

List<Order> orders = orderRepository.findAll(); // SELECT * FROM orders
orders.forEach(o -> o.getUser().getName());     // 각 order마다 SELECT * FROM users WHERE id=?

    // 해결 1: Fetch Join (JPQL)
@Query("SELECT o FROM Order o JOIN FETCH o.user WHERE o.status = :status")
List<Order> findByStatusWithUser(@Param("status") String status);
// 실행: SELECT o.*, u.* FROM orders o INNER JOIN users u ON o.user_id = u.id

    // 해결 2: EntityGraph
@EntityGraph(attributePaths = {"user"})
List<Order> findByStatus(String status);

    // 해결 3: BatchSize (컬렉션 N+1 해결)
@Entity
public class User {
    @OneToMany(mappedBy = "user")
    @BatchSize(size = 100) // 100개씩 IN 쿼리로 처리
    private List<Order> orders;
}
// SELECT * FROM orders WHERE user_id IN (1, 2, 3, ... 100)

    # 전역 설정
spring:
  jpa:
    properties:
      hibernate:
        default_batch_fetch_size: 100

    3. 페이징 처리 최적화

    -- Offset 방식의 문제: OFFSET이 클수록 느려짐
SELECT * FROM orders ORDER BY id DESC LIMIT 10 OFFSET 1000000;
-- MySQL은 1000010개를 읽고 앞의 1000000개를 버림!

-- 커버링 인덱스 + 서브쿼리로 최적화
SELECT o.*
FROM orders o
JOIN (
    SELECT id FROM orders ORDER BY id DESC LIMIT 1000000, 10
) sub ON o.id = sub.id;
-- 내부 서브쿼리는 커버링 인덱스(id만 조회)로 빠르게 처리

    // Cursor 기반 페이징 (No Offset)
@Query("""
    SELECT o FROM Order o
    WHERE o.id < :lastId
    ORDER BY o.id DESC
    LIMIT :size
    """)
List<Order> findByCursor(@Param("lastId") Long lastId, @Param("size") int size);

    4. 커버링 인덱스 (Covering Index)

    -- 쿼리에 필요한 모든 컬럼이 인덱스에 포함된 경우
-- 테이블 접근 없이 인덱스만으로 결과 반환 → 매우 빠름

CREATE INDEX idx_orders_covering ON orders(user_id, status, created_at, total_amount);

-- 아래 쿼리는 테이블 접근 없이 인덱스만으로 처리
SELECT user_id, status, created_at, total_amount
FROM orders
WHERE user_id = 1 AND status = 'PAID';
-- EXPLAIN Extra: "Using index" → 커버링 인덱스 사용

    5. 파티셔닝

    -- 날짜 기반 Range 파티셔닝 (로그, 주문 등 시계열 데이터)
CREATE TABLE orders_partitioned (
    id BIGINT NOT NULL,
    user_id BIGINT NOT NULL,
    status VARCHAR(20),
    created_at DATETIME NOT NULL,
    PRIMARY KEY (id, created_at)
)
PARTITION BY RANGE (YEAR(created_at)) (
    PARTITION p2024 VALUES LESS THAN (2025),
    PARTITION p2025 VALUES LESS THAN (2026),
    PARTITION p2026 VALUES LESS THAN (2027),
    PARTITION p_future VALUES LESS THAN MAXVALUE
);

-- 2026년 데이터만 조회 시 해당 파티션만 스캔
SELECT * FROM orders_partitioned WHERE created_at >= '2026-01-01';

    6. 읽기 전용 복제본(Read Replica) 분리

    # application.yml - DataSource Routing 설정
spring:
  datasource:
    primary:
      url: jdbc:mysql://master-db:3306/mydb
    replica:
      url: jdbc:mysql://replica-db:3306/mydb

    // @Transactional(readOnly=true) 쿼리를 자동으로 Replica로 라우팅
@Service
public class OrderService {

    @Transactional(readOnly = true) // → Replica DB 사용
    public List<Order> findOrders(Long userId) {
        return orderRepository.findByUserId(userId);
    }

    @Transactional // → Primary DB 사용
    public Order createOrder(OrderRequest request) {
        return orderRepository.save(new Order(request));
    }
}

    실무 최적화 사례

    사례 1: 페이지네이션 슬로우 쿼리 개선

    -- 문제 쿼리 (10초 이상 소요)
SELECT * FROM products WHERE category_id = 5 ORDER BY created_at DESC LIMIT 10 OFFSET 500000;

-- 1단계: EXPLAIN 확인
EXPLAIN SELECT * FROM products WHERE category_id = 5 ORDER BY created_at DESC LIMIT 10 OFFSET 500000;
-- type: ALL, rows: 5000000 → Full Scan!

-- 2단계: 복합 인덱스 추가
CREATE INDEX idx_products_category_created ON products(category_id, created_at DESC);

-- 3단계: No-Offset 페이징으로 변경
SELECT * FROM products
WHERE category_id = 5 AND created_at < '2026-03-20 10:00:00'
ORDER BY created_at DESC
LIMIT 10;
-- 결과: 10ms 미만으로 개선

    사례 2: N+1 쿼리로 인한 장애

    // 문제: 주문 목록 API에서 N+1 발생
@GetMapping("/orders")
public List<OrderResponse> getOrders() {
    List<Order> orders = orderRepository.findAll(); // 1번 쿼리
    return orders.stream()
        .map(order -> new OrderResponse(
            order.getId(),
            order.getUser().getName(),  // N번 쿼리!
            order.getStatus()
        ))
        .toList();
}
// 주문 1만건 → 1만 1번 쿼리 → DB 과부하 → 장애

    // 해결: Fetch Join으로 한 번에 조회
@Query("SELECT o FROM Order o JOIN FETCH o.user")
List<Order> findAllWithUser();

// 또는 DTO로 바로 조회 (불필요한 엔티티 로드 X)
@Query("""
    SELECT new com.example.OrderDto(o.id, u.name, o.status)
    FROM Order o JOIN o.user u
    """)
List<OrderDto> findAllOrderDto();

    사례 3: LIKE 검색 최적화

    -- 문제: 상품명 검색이 느림
SELECT * FROM products WHERE name LIKE '%무선이어폰%';
-- type: ALL, rows: 1000000

-- 해결 1: Full-Text Index
ALTER TABLE products ADD FULLTEXT INDEX ft_name(name) WITH PARSER ngram;
SELECT * FROM products WHERE MATCH(name) AGAINST('무선이어폰' IN BOOLEAN MODE);

-- 해결 2: Elasticsearch 도입 (실무 추천)
-- 복잡한 검색, 형태소 분석, 자동완성이 필요하면 Elasticsearch 사용

    쿼리 튜닝 체크리스트

    항목 확인 방법 주의점
    Full Table Scan 여부 EXPLAIN type = ALL 확인 1만 건 이상 테이블에서 ALL은 위험
    인덱스 사용 여부 EXPLAIN key 컬럼 확인 NULL이면 인덱스 미사용
    조회 행 수 EXPLAIN rows 컬럼 확인 실제 결과 대비 너무 많으면 비효율
    Using filesort EXPLAIN Extra 확인 ORDER BY에 인덱스 미사용
    Using temporary EXPLAIN Extra 확인 임시 테이블 생성 (GROUP BY 최적화 필요)
    N+1 여부 쿼리 로그 확인 같은 쿼리 N번 반복 실행
    SELECT * 사용 코드 리뷰 필요한 컬럼만 조회
    OFFSET 크기 쿼리 분석 대규모 OFFSET → Cursor 방식으로 전환
    트랜잭션 범위 코드 리뷰 불필요하게 긴 트랜잭션 내 슬로우 쿼리
    인덱스 과다 SHOW INDEX 확인 인덱스 많으면 쓰기 성능 저하

    인덱스 설계 원칙 정리

    인덱스를 추가해야 할 컬럼:
- WHERE 절에서 자주 사용되는 컬럼
- JOIN ON 조건에 사용되는 컬럼
- ORDER BY, GROUP BY에 사용되는 컬럼
- 카디널리티(유니크 값 비율)가 높은 컬럼 우선

인덱스를 피해야 할 컬럼:
- 자주 UPDATE되는 컬럼 (쓰기 성능 저하)
- 카디널리티가 매우 낮은 컬럼 (성별 Y/N 등 → 인덱스 효과 미미)
- 테이블이 아주 작은 경우 (Full Scan이 더 빠를 수 있음)

복합 인덱스 컬럼 순서 원칙:
1. 동등(=) 조건 컬럼 먼저
2. 카디널리티 높은 컬럼 먼저
3. 범위(<, >) 조건 컬럼은 마지막

    서비스 초기 (데이터 적음) 기본 인덱스만 설정 (과도한 최적화 지양)

    페이지네이션 느려짐 No-Offset (Cursor) 방식 전환
    목록 API 응답 느림 N+1 점검, Fetch Join / BatchSize 적용
    검색 기능 느림 Full-Text Index 또는 Elasticsearch 도입
    특정 시간대 DB 과부하 Read Replica 분리, 캐싱 레이어(Redis) 추가
    대량 배치 처리 청크 단위 처리, 인덱스 임시 비활성화 검토
