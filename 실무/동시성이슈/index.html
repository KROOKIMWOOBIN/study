    лҸҷмӢңм„ұ мқҙмҠҲ (Concurrency Issues)

    мҷң кіөл¶Җн•ҳлҠ”к°Җ?

    мӢӨл¬ҙм—җм„ң мһ¬кі  м°Ёк°җ, нҸ¬мқёнҠё м ҒлҰҪ/м°Ёк°җ, мўӢм•„мҡ” м№ҙмҡҙнҠё, м„ м°©мҲң мқҙлІӨнҠё мІҳлҰ¬ л“ұ м—¬лҹ¬ мӮ¬мҡ©мһҗк°Җ лҸҷмӢңм—җ к°ҷмқҖ мһҗмӣҗм—җ м ‘к·јн•ҳлҠ” мғҒнҷ©мқҖ л§Өмҡ° нқ”н•ҳлӢӨ. мқҙлҘј м ңлҢҖлЎң мІҳлҰ¬н•ҳм§Җ м•Ҡмңјл©ҙ лҚ°мқҙн„° м •н•©м„ұмқҙ к№Ём§ҖлҠ” мӢ¬к°Ғн•ң лІ„к·ёк°Җ л°ңмғқн•ңлӢӨ.

    лҸҷмӢңм„ұ л¬ём ңлһҖ?

    м—¬лҹ¬ мҠӨл Ҳл“ң лҳҗлҠ” н”„лЎңм„ёмҠӨк°Җ лҸҷмӢңм—җ кіөмң  мһҗмӣҗ(л©”лӘЁлҰ¬, DB л ҲмҪ”л“ң л“ұ)м—җ м ‘к·јн•  л•Ң мӢӨн–ү мҲңм„ңм—җ л”°лқј кІ°кіјк°Җ лӢ¬лқјм§ҖлҠ” л¬ём ң.

    Thread A: мһ¬кі  мқҪкё°(10) вҶ’ м°Ёк°җ м—°мӮ°(10-1=9) вҶ’ м ҖмһҘ(9)
Thread B: мһ¬кі  мқҪкё°(10) вҶ’ м°Ёк°җ м—°мӮ°(10-1=9) вҶ’ м ҖмһҘ(9)

кё°лҢҖ кІ°кіј: 8
мӢӨм ң кІ°кіј: 9  вҶҗ лҚ°мқҙн„° м •н•©м„ұ мҳӨлҘҳ!

    Race Condition (кІҪмҹҒ мЎ°кұҙ)

    м •мқҳ

    л‘җ к°ң мқҙмғҒмқҳ мҠӨл Ҳл“ңк°Җ кіөмң  мһҗмӣҗм—җ лҸҷмӢңм—җ м ‘к·јн•ҳм—¬ мқҪкё°-мҲҳм •-м“°кё°(Read-Modify-Write) лҘј мҲҳн–үн•  л•Ң, мӢӨн–ү мҲңм„ңм—җ л”°лқј кІ°кіјк°Җ лӢ¬лқјм§ҖлҠ” нҳ„мғҒ.

    л°ңмғқн•ҳлҠ” лҢҖн‘ң мӢңлӮҳлҰ¬мҳӨ

    мӢңлӮҳлҰ¬мҳӨ мҰқмғҒ
    мһ¬кі  м°Ёк°җ мһ¬кі к°Җ 0 мқҙн•ҳлЎң лӮҙл Өк°җ (мҳӨлІ„м…Җл§Ғ)
    нҸ¬мқёнҠё м°Ёк°җ мһ”кі ліҙлӢӨ л§Һмқҙ м°Ёк°җлҗЁ
    мўӢм•„мҡ” м№ҙмҡҙнҠё лҸҷмӢң нҒҙлҰӯ мӢң м№ҙмҡҙнҠёк°Җ лҲ„лқҪлҗЁ
    м„ м°©мҲң мҝ нҸ° л°ңкёү м •н•ҙм§„ мҲҳлҹүліҙлӢӨ л§Һмқҙ л°ңкёүлҗЁ
    нҡҢмӣҗ к°Җмһ… мӨ‘ліө к°ҷмқҖ мқҙл©”мқјлЎң мӨ‘ліө к°Җмһ…лҗЁ

    Javaм—җм„ң Race Condition л°ңмғқ мҳҲмӢң

    // мһ¬кі  м°Ёк°җ - Race Condition л°ңмғқ мҪ”л“ң
@Service
public class StockService {
    private int stock = 100; // кіөмң  мһҗмӣҗ

    public void decrease() {
        if (stock <= 0) {
            throw new IllegalStateException("мһ¬кі  л¶ҖмЎұ");
        }
        stock--; // мӣҗмһҗм Ғ м—°мӮ°мқҙ м•„лӢҳ! (мқҪкё° вҶ’ к°җмҶҢ вҶ’ м“°кё°)
    }
}

    // н…ҢмҠӨнҠё: 100к°ң мҠӨл Ҳл“ңк°Җ лҸҷмӢңм—җ м°Ёк°җ
@Test
void raceConditionTest() throws InterruptedException {
    int threadCount = 100;
    ExecutorService executorService = Executors.newFixedThreadPool(32);
    CountDownLatch latch = new CountDownLatch(threadCount);

    for (int i = 0; i < threadCount; i++) {
        executorService.submit(() -> {
            try {
                stockService.decrease();
            } finally {
                latch.countDown();
            }
        });
    }
    latch.await();

    // кё°лҢҖк°’: 0, мӢӨм ңк°’: 0мқҙ м•„лӢҗ к°ҖлҠҘм„ұ лҶ’мқҢ
    System.out.println("мөңмў… мһ¬кі : " + stockService.getStock());
}

    н•ҙкІ° л°©лІ•

    1. synchronized (лӢЁмқј м„ңлІ„, JVM л ҲлІЁ)

    мҷң мӮ¬мҡ©н•ҳлҠ”к°Җ? JVM лӮҙм—җм„ң н•ң лІҲм—җ н•ҳлӮҳмқҳ мҠӨл Ҳл“ңл§Ң н•ҙлӢ№ л©”м„ңл“ң/лё”лЎқм—җ м ‘к·јн•ҳлҸ„лЎқ к°•м ңн•ңлӢӨ.

    м–ҙл–»кІҢ мӮ¬мҡ©н•ҳлҠ”к°Җ?

    @Service
public class StockService {
    private int stock = 100;

    // л©”м„ңл“ң л ҲлІЁ лҸҷкё°нҷ”
    public synchronized void decrease() {
        if (stock <= 0) {
            throw new IllegalStateException("мһ¬кі  л¶ҖмЎұ");
        }
        stock--;
    }

    // лё”лЎқ л ҲлІЁ лҸҷкё°нҷ” (лҚ” м„ёл°Җн•ң м ңм–ҙ)
    public void decreaseBlock() {
        synchronized (this) {
            if (stock <= 0) {
                throw new IllegalStateException("мһ¬кі  л¶ҖмЎұ");
            }
            stock--;
        }
    }
}

    мЈјмқҳ: @Transactionalкіј н•Ёк»ҳ м“ё л•Ң л¬ём ң л°ңмғқ к°ҖлҠҘ Springмқҳ @TransactionalмқҖ н”„лЎқмӢң кё°л°ҳмқҙлқј synchronizedк°Җ нҠёлһңмһӯм…ҳ м»Өл°Ӣ м „м—җ н•ҙм ңлҗ  мҲҳ мһҲлӢӨ.

    мһҘм җ - кө¬нҳ„ к°„лӢЁ - м¶”к°Җ мқҳмЎҙм„ұ м—ҶмқҢ

    лӢЁм җ - лӢӨмӨ‘ м„ңлІ„(л¶„мӮ° нҷҳкІҪ)м—җм„ң лҸҷмһ‘н•ҳм§Җ м•ҠмқҢ - м„ұлҠҘ м Җн•ҳ (лі‘лӘ©) - @Transactionalкіј н•Ёк»ҳ мӮ¬мҡ© мӢң мЈјмқҳ н•„мҡ”

    м–ҙл–Ё л•Ң м“°лҠ”к°Җ? лӢЁмқј м„ңлІ„, к°„лӢЁн•ң м№ҙмҡҙн„°, лӮҙл¶Җ мәҗмӢң л“ұ JVM лӮҙм—җм„ңл§Ң кіөмң лҗҳлҠ” мһҗмӣҗ.

    2. ReentrantLock (лӢЁмқј м„ңлІ„, лҚ” мң м—°н•ң лқҪ)

    мҷң мӮ¬мҡ©н•ҳлҠ”к°Җ? synchronizedліҙлӢӨ лҚ” м„ёл°Җн•ң м ңм–ҙ (нғҖмһ„м•„мӣғ, кіөм •м„ұ м„Өм •, tryLock л“ұ)к°Җ н•„мҡ”н•  л•Ң.

    @Service
public class StockService {
    private int stock = 100;
    private final ReentrantLock lock = new ReentrantLock(true); // true = кіөм • лқҪ

    public void decrease() {
        lock.lock();
        try {
            if (stock <= 0) {
                throw new IllegalStateException("мһ¬кі  л¶ҖмЎұ");
            }
            stock--;
        } finally {
            lock.unlock(); // л°ҳл“ңмӢң finallyм—җм„ң н•ҙм ң
        }
    }

    // нғҖмһ„м•„мӣғмқҙ мһҲлҠ” лқҪ нҡҚл“қ мӢңлҸ„
    public boolean tryDecrease() throws InterruptedException {
        if (lock.tryLock(3, TimeUnit.SECONDS)) {
            try {
                stock--;
                return true;
            } finally {
                lock.unlock();
            }
        }
        return false; // 3мҙҲ лӮҙ лқҪ нҡҚл“қ мӢӨнҢЁ
    }
}

    мһҘм җ - tryLockмңјлЎң лҚ°л“ңлқҪ л°©м§Җ - кіөм • лқҪ(fairness) м„Өм • к°ҖлҠҘ - мЎ°кұҙ ліҖмҲҳ(Condition) мӮ¬мҡ© к°ҖлҠҘ

    лӢЁм җ - synchronizedліҙлӢӨ мҪ”л“ңк°Җ ліөмһЎ - unlock()мқ„ finallyм—җм„ң л°ҳл“ңмӢң нҳём¶ңн•ҙм•ј н•Ё - л¶„мӮ° нҷҳкІҪ лҜём§Җмӣҗ

    3. Atomic нҒҙлһҳмҠӨ (AtomicInteger, AtomicLong)

    мҷң мӮ¬мҡ©н•ҳлҠ”к°Җ? CAS(Compare-And-Swap) м—°мӮ°мқ„ мӮ¬мҡ©н•ҙ лқҪ м—Ҷмқҙ мӣҗмһҗм Ғ м—°мӮ°мқ„ ліҙмһҘн•ңлӢӨ. м„ұлҠҘмқҙ мўӢлӢӨ.

    @Service
public class StockService {
    private AtomicInteger stock = new AtomicInteger(100);

    public void decrease() {
        int current;
        do {
            current = stock.get();
            if (current <= 0) {
                throw new IllegalStateException("мһ¬кі  л¶ҖмЎұ");
            }
        } while (!stock.compareAndSet(current, current - 1));
        // CAS: currentк°’мқҙ ліҖкІҪлҗҳм§Җ м•Ҡм•ҳмңјл©ҙ current-1лЎң м„Өм •
    }

    // лҚ” к°„лӢЁн•ң л°©лІ•
    public void simpleDecrease() {
        if (stock.decrementAndGet() < 0) {
            stock.incrementAndGet(); // лЎӨл°ұ
            throw new IllegalStateException("мһ¬кі  л¶ҖмЎұ");
        }
    }
}

    мһҘм җ - лқҪліҙлӢӨ м„ұлҠҘмқҙ мўӢмқҢ (Lock-free) - кө¬нҳ„ к°„лӢЁ

    лӢЁм җ - л¶„мӮ° нҷҳкІҪ лҜём§Җмӣҗ - ліөмһЎн•ң л№„мҰҲлӢҲмҠӨ лЎңм§Ғм—” л¶Җм Ғн•© (лӢЁмҲң м№ҙмҡҙн„°м—җ м Ғн•©) - ABA л¬ём ң л°ңмғқ к°ҖлҠҘ (AtomicStampedReferenceлЎң н•ҙкІ°)

    4. DB л№„кҙҖм Ғ лқҪ (Pessimistic Lock, SELECT FOR UPDATE)

    мҷң мӮ¬мҡ©н•ҳлҠ”к°Җ? м¶©лҸҢмқҙ мһҗмЈј л°ңмғқн•ҳлҠ” нҷҳкІҪм—җм„ң лҚ°мқҙн„°лҘј мқҪлҠ” мӢңм җм—җ лқҪмқ„ кұём–ҙ лӢӨлҘё нҠёлһңмһӯм…ҳмқҳ м ‘к·јмқ„ л§үлҠ”лӢӨ.

    м–ҙл–»кІҢ мӮ¬мҡ©н•ҳлҠ”к°Җ?

    // Repository
public interface StockRepository extends JpaRepository<Stock, Long> {

    @Lock(LockModeType.PESSIMISTIC_WRITE)
    @Query("SELECT s FROM Stock s WHERE s.id = :id")
    Optional<Stock> findByIdWithPessimisticLock(@Param("id") Long id);
}

    -- мӢӨм ң мӢӨн–үлҗҳлҠ” SQL
SELECT * FROM stock WHERE id = 1 FOR UPDATE;
-- лӢӨлҘё нҠёлһңмһӯм…ҳмқҖ н•ҙлӢ№ л ҲмҪ”л“ңмқҳ лқҪмқҙ н•ҙм ңлҗ  л•Ңк№Ңм§Җ лҢҖкё°

    @Service
@RequiredArgsConstructor
public class StockService {
    private final StockRepository stockRepository;

    @Transactional
    public void decrease(Long stockId, Long quantity) {
        Stock stock = stockRepository.findByIdWithPessimisticLock(stockId)
            .orElseThrow(() -> new EntityNotFoundException("мһ¬кі  м—ҶмқҢ"));
        stock.decrease(quantity);
    }
}

    мһҘм җ - м¶©лҸҢмқҙ л§ҺмқҖ нҷҳкІҪм—җм„ң мӢ лў°м„ұ лҶ’мқҢ - лҚ°мқҙн„° м •н•©м„ұ ліҙмһҘмқҙ нҷ•мӢӨн•Ё

    лӢЁм җ - м„ұлҠҘ м Җн•ҳ (лқҪ лҢҖкё° мӢңк°„ л°ңмғқ) - лҚ°л“ңлқҪ мң„н—ҳ - л¶„мӮ° DB нҷҳкІҪм—җм„ңлҠ” м„ұлҠҘ л¬ём ң мӢ¬нҷ”

    м–ҙл–Ё л•Ң м“°лҠ”к°Җ? мһ¬кі  м°Ёк°җмІҳлҹј м¶©лҸҢ к°ҖлҠҘм„ұмқҙ лҶ’кі , м •н•©м„ұмқҙ л§Өмҡ° мӨ‘мҡ”н•ң кІҪмҡ°.

    5. DB лӮҷкҙҖм Ғ лқҪ (Optimistic Lock, @Version)

    мҷң мӮ¬мҡ©н•ҳлҠ”к°Җ? м¶©лҸҢмқҙ м ҒмқҖ нҷҳкІҪм—җм„ң лІ„м „ лІҲнҳёлҘј мқҙмҡ©н•ҙ лҸҷмӢң мҲҳм •мқ„ к°җм§Җн•ңлӢӨ. мӢӨм ң лқҪмқ„ кұём§Җ м•Ҡм•„ м„ұлҠҘмқҙ мўӢлӢӨ.

    @Entity
public class Stock {
    @Id
    @GeneratedValue
    private Long id;

    private Long quantity;

    @Version // лӮҷкҙҖм Ғ лқҪмқҳ н•өмӢ¬: м—…лҚ°мқҙнҠё мӢңл§ҲлӢӨ мһҗлҸҷ мҰқк°Җ
    private Long version;

    public void decrease(Long quantity) {
        if (this.quantity < quantity) {
            throw new IllegalStateException("мһ¬кі  л¶ҖмЎұ");
        }
        this.quantity -= quantity;
    }
}

    // Repository
public interface StockRepository extends JpaRepository<Stock, Long> {

    @Lock(LockModeType.OPTIMISTIC)
    @Query("SELECT s FROM Stock s WHERE s.id = :id")
    Optional<Stock> findByIdWithOptimisticLock(@Param("id") Long id);
}

    // м„ңл№„мҠӨ: м¶©лҸҢ мӢң мһ¬мӢңлҸ„ лЎңм§Ғ н•„мҡ”
@Service
@RequiredArgsConstructor
public class OptimisticLockStockFacade {
    private final StockRepository stockRepository;

    public void decrease(Long id, Long quantity) throws InterruptedException {
        while (true) {
            try {
                stockRepository.findByIdWithOptimisticLock(id)
                    .ifPresent(stock -> {
                        stock.decrease(quantity);
                        stockRepository.save(stock);
                    });
                break;
            } catch (OptimisticLockingFailureException e) {
                // лІ„м „ м¶©лҸҢ мӢң мһ¬мӢңлҸ„
                Thread.sleep(50);
            }
        }
    }
}

UPDATE stock SET quantity = 9, version = 2 WHERE id = 1 AND version = 1;

    мһҘм җ - лқҪмқ„ кұём§Җ м•Ҡм•„ м„ұлҠҘ мўӢмқҢ - м¶©лҸҢмқҙ м ҒмқҖ нҷҳкІҪм—җ м Ғн•©

    лӢЁм җ - м¶©лҸҢ мӢң мһ¬мӢңлҸ„ лЎңм§Ғмқ„ м§Ғм ‘ кө¬нҳ„н•ҙм•ј н•Ё - м¶©лҸҢмқҙ мһҰмңјл©ҙ мһ¬мӢңлҸ„к°Җ л§Һм•„м ё м„ұлҠҘ м Җн•ҳ

    м–ҙл–Ё л•Ң м“°лҠ”к°Җ? лҸҷмӢң мҲҳм • к°ҖлҠҘм„ұмқҙ лӮ®кі , м¶©лҸҢ мӢң мӢӨнҢЁн•ҙлҸ„ кҙңм°®мқҖ кІҪмҡ° (мҳҲ: кІҢмӢңкёҖ мҲҳм •).

    6. Redis л¶„мӮ° лқҪ (Distributed Lock)

    мҷң мӮ¬мҡ©н•ҳлҠ”к°Җ? лӢӨмӨ‘ м„ңлІ„(л¶„мӮ° нҷҳкІҪ)м—җм„ң JVM л ҲлІЁмқҳ лқҪмқҖ нҡЁкіјк°Җ м—ҶлӢӨ. RedisлҘј мӨ‘м•ҷ лқҪ м ҖмһҘмҶҢлЎң мӮ¬мҡ©н•ҳл©ҙ м—¬лҹ¬ м„ңлІ„ к°„ лҸҷкё°нҷ”к°Җ к°ҖлҠҘн•ҳлӢӨ.

    Redisson мӮ¬мҡ© (м¶”мІң)

    // build.gradle
// implementation 'org.redisson:redisson-spring-boot-starter:3.23.4'

@Service
@RequiredArgsConstructor
public class RedissonLockStockService {
    private final RedissonClient redissonClient;
    private final StockRepository stockRepository;

    public void decrease(Long key, Long quantity) throws InterruptedException {
        RLock lock = redissonClient.getLock("stock:" + key);

        try {
            // лқҪ нҡҚл“қ мӢңлҸ„: мөңлҢҖ 5мҙҲ лҢҖкё°, лқҪ мң м§Җ мӢңк°„ 3мҙҲ
            boolean available = lock.tryLock(5, 3, TimeUnit.SECONDS);
            if (!available) {
                throw new RuntimeException("лқҪ нҡҚл“қ мӢӨнҢЁ");
            }

            Stock stock = stockRepository.findById(key)
                .orElseThrow(() -> new EntityNotFoundException("мһ¬кі  м—ҶмқҢ"));
            stock.decrease(quantity);
            stockRepository.save(stock);

        } finally {
            lock.unlock(); // л°ҳл“ңмӢң н•ҙм ң
        }
    }
}

    Lettuce мӮ¬мҡ© (Spin Lock л°©мӢқ)

    // LettuceлҠ” кё°ліём ҒмңјлЎң spin lock (CPU лӮӯл№„) л°©мӢқ
@Component
@RequiredArgsConstructor
public class RedisLockRepository {
    private final RedisTemplate<String, String> redisTemplate;

    public Boolean lock(Long key) {
        return redisTemplate.opsForValue()
            .setIfAbsent("lock:" + key, "lock", Duration.ofMillis(3_000));
    }

    public Boolean unlock(Long key) {
        return redisTemplate.delete("lock:" + key);
    }
}

@Service
@RequiredArgsConstructor
public class LettuceLockStockFacade {
    private final RedisLockRepository redisLockRepository;
    private final StockRepository stockRepository;

    public void decrease(Long id, Long quantity) throws InterruptedException {
        while (!redisLockRepository.lock(id)) {
            Thread.sleep(100); // spin lock: Redis л¶Җн•ҳ мЈјмқҳ
        }
        try {
            Stock stock = stockRepository.findById(id)
                .orElseThrow();
            stock.decrease(quantity);
            stockRepository.save(stock);
        } finally {
            redisLockRepository.unlock(id);
        }
    }
}
