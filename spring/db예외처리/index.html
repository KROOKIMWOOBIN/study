    мҳҲмҷё ліҖнҷҳкё° (SQLExceptionTranslator)

    м—ӯн• 

    SQLExceptionмқҳ м—җлҹ¬ мҪ”л“ңлҘј мҠӨн”„л§Ғ DataAccessException кі„мёөмңјлЎң ліҖнҷҳн•ңлӢӨ.

    DataAccessException translate(String task, String sql, SQLException ex);

    SQLErrorCodeSQLExceptionTranslator

    мҠӨн”„л§Ғмқҙ кё°ліёмңјлЎң м ңкіөн•ҳлҠ” кө¬нҳ„мІҙ. DBлі„ м—җлҹ¬ мҪ”л“ң л§Өн•‘ нҢҢмқјмқ„ к°Җм§Җкі  мһҲлӢӨ.

    @Slf4j
public class MemberRepositoryV4 implements MemberRepository {

    private final DataSource dataSource;
    private final SQLExceptionTranslator exceptionTranslator;

    public MemberRepositoryV4(DataSource dataSource) {
        this.dataSource = dataSource;
        // мғқм„ұмһҗм—җ DataSource м „лӢ¬ вҶ’ DB мў…лҘҳ мһҗлҸҷ к°җм§Җ
        this.exceptionTranslator = new SQLErrorCodeSQLExceptionTranslator(dataSource);
    }

    @Override
    public Member save(Member member) {
        String sql = "INSERT INTO member(member_id, money) VALUES(?, ?)";
        Connection con = null;
        PreparedStatement pstmt = null;
        try {
            con = getConnection();
            pstmt = con.prepareStatement(sql);
            pstmt.setString(1, member.getMemberId());
            pstmt.setInt(2, member.getMoney());
            pstmt.executeUpdate();
            return member;
        } catch (SQLException e) {
            // SQLExceptionмқ„ мҠӨн”„л§Ғ мҳҲмҷёлЎң ліҖнҷҳ!
            throw exceptionTranslator.translate("save", sql, e);
        } finally {
            close(con, pstmt, null);
        }
    }
}

    Spring JdbcTemplateкіј мҳҲмҷё мІҳлҰ¬

    Spring JdbcTemplateмқҖ лӮҙл¶Җм ҒмңјлЎң мҳҲмҷё ліҖнҷҳмқ„ мһҗлҸҷмңјлЎң мІҳлҰ¬н•ңлӢӨ.

    @Repository
public class MemberRepository {

    private final JdbcTemplate jdbcTemplate;

    public MemberRepository(DataSource dataSource) {
        this.jdbcTemplate = new JdbcTemplate(dataSource);
    }

    public Member save(Member member) {
        String sql = "INSERT INTO member(member_id, money) VALUES(?, ?)";
        // JdbcTemplateмқҙ SQLException вҶ’ DataAccessException мһҗлҸҷ ліҖнҷҳ
        // try-catch, м»Өл„Ҙм…ҳ кҙҖлҰ¬, мһҗмӣҗ м •лҰ¬ лӘЁл‘җ JdbcTemplateмқҙ мІҳлҰ¬
        jdbcTemplate.update(sql, member.getMemberId(), member.getMoney());
        return member;
    }

    public Member findById(String memberId) {
        String sql = "SELECT member_id, money FROM member WHERE member_id = ?";
        return jdbcTemplate.queryForObject(sql, memberRowMapper(), memberId);
    }

    private RowMapper<Member> memberRowMapper() {
        return (rs, rowNum) -> {
            Member member = new Member();
            member.setMemberId(rs.getString("member_id"));
            member.setMoney(rs.getInt("money"));
            return member;
        };
    }
}
