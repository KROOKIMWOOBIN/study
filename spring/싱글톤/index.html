    мӢұкёҖнҶӨ (Singleton)

    мӢұкёҖнҶӨ нҢЁн„ҙмқҙлһҖ?

    нҒҙлһҳмҠӨмқҳ мқёмҠӨн„ҙмҠӨк°Җ л”ұ 1к°ңл§Ң мғқм„ұлҗҳлҠ” кІғмқ„ ліҙмһҘн•ҳлҠ” л””мһҗмқё нҢЁн„ҙ.

    мҷң мӢұкёҖнҶӨмқҙ н•„мҡ”н•ңк°Җ?

    мӣ№ м• н”ҢлҰ¬мјҖмқҙм…ҳм—җм„ң мҲҳл§ҺмқҖ мҡ”мІӯмқҙ л“Өм–ҙмҳ¬ л•Ңл§ҲлӢӨ к°қмІҙлҘј мғҲлЎң мғқм„ұн•ҳл©ҙ:

    нҒҙлқјмқҙм–ёнҠё A мҡ”мІӯ вҶ’ MemberService мғқм„ұ (к°қмІҙ 1к°ң)
нҒҙлқјмқҙм–ёнҠё B мҡ”мІӯ вҶ’ MemberService мғқм„ұ (к°қмІҙ 1к°ң)
нҒҙлқјмқҙм–ёнҠё C мҡ”мІӯ вҶ’ MemberService мғқм„ұ (к°қмІҙ 1к°ң)
...
мҙҲлӢ№ 100к°ң мҡ”мІӯ вҶ’ 100к°ңмқҳ MemberService мғқм„ұ

    л©”лӘЁлҰ¬ лӮӯл№„к°Җ мӢ¬к°Ғн•ҙм§„лӢӨ. лҢҖл¶Җл¶„мқҳ м„ңл№„мҠӨ к°қмІҙлҠ” мғҒнғң м—Ҷмқҙ(Stateless) мӮ¬мҡ©н•ҳлҜҖлЎң н•ҳлӮҳл§Ң л§Ңл“Өм–ҙм„ң кіөмң н•ҳлҠ” кІғмқҙ н•©лҰ¬м ҒмқҙлӢӨ.

    мҲңмҲҳ мһҗл°” мӢұкёҖнҶӨ нҢЁн„ҙ кө¬нҳ„

    public class SingletonService {

    // 1. static н•„л“ңм—җ л”ұ 1к°ңл§Ң мғқм„ұ
    private static final SingletonService instance = new SingletonService();

    // 2. мҷёл¶Җм—җм„ң new л¶Ҳк°Җ
    private SingletonService() {}

    // 3. мқёмҠӨн„ҙмҠӨлҘј л°ҳнҷҳн•ҳлҠ” static л©”м„ңл“ң
    public static SingletonService getInstance() {
        return instance;
    }

    public void logic() {
        System.out.println("мӢұкёҖнҶӨ к°қмІҙ лЎңм§Ғ нҳём¶ң");
    }
}

    // н…ҢмҠӨнҠё
SingletonService s1 = SingletonService.getInstance();
SingletonService s2 = SingletonService.getInstance();
assertThat(s1).isSameAs(s2); // к°ҷмқҖ мқёмҠӨн„ҙмҠӨ!

    мҲңмҲҳ мӢұкёҖнҶӨ нҢЁн„ҙмқҳ л¬ём ңм җ

    л¬ём ң м„ӨлӘ…
    кө¬нҳ„ мҪ”л“ң мҰқк°Җ getInstance(), private мғқм„ұмһҗ л“ұ л¶Җк°Җ мҪ”л“ң мһ‘м„ұ н•„мҡ”
    DIP мң„л°ҳ нҒҙлқјмқҙм–ёнҠёк°Җ getInstance()лҘј м§Ғм ‘ нҳём¶ң вҶ’ кө¬мІҙ нҒҙлһҳмҠӨм—җ мқҳмЎҙ
    OCP мң„л°ҳ ліҖкІҪмқҙ м–ҙл ӨмӣҖ
    н…ҢмҠӨнҠё м–ҙл ӨмӣҖ Mock көҗмІҙ л¶Ҳк°Җ, н…ҢмҠӨнҠё к°„ мғҒнғң кіөмң лЎң кІ©лҰ¬ м–ҙл ӨмӣҖ
    мң м—°м„ұ л¶ҖмЎұ мһҗмӢқ нҒҙлһҳмҠӨ л§Ңл“Өкё° м–ҙл ӨмӣҖ
    л©ҖнӢ°мҠӨл Ҳл“ң мң„н—ҳ кө¬нҳ„ л°©мӢқм—җ л”°лқј мҠӨл Ҳл“ң м•Ҳм „н•ҳм§Җ м•Ҡмқ„ мҲҳ мһҲмқҢ 
    // DIP мң„л°ҳ мҳҲмӢң
public class ClientA {
    // getInstance()лҘј нҶөн•ҙ кө¬мІҙ нҒҙлһҳмҠӨлҘј м§Ғм ‘ м°ёмЎ°
    private SingletonService service = SingletonService.getInstance();
}

    мҠӨн”„л§Ғмқҳ мӢұкёҖнҶӨ м»Ён…Ңмқҙл„Ҳ (мӢұкёҖнҶӨ л Ҳм§ҖмҠӨнҠёлҰ¬)

    мҠӨн”„л§Ғ м»Ён…Ңмқҙл„ҲлҠ” мӢұкёҖнҶӨ нҢЁн„ҙмқҳ лӢЁм җмқ„ н•ҙкІ°н•ҳл©ҙм„ң мӢұкёҖнҶӨмңјлЎң к°қмІҙлҘј кҙҖлҰ¬н•ңлӢӨ.

    ApplicationContext ac = new AnnotationConfigApplicationContext(AppConfig.class);

MemberService s1 = ac.getBean("memberService", MemberService.class);
MemberService s2 = ac.getBean("memberService", MemberService.class);

// мҠӨн”„л§Ғ м»Ён…Ңмқҙл„Ҳк°Җ к°ҷмқҖ мқёмҠӨн„ҙмҠӨ л°ҳнҷҳ!
assertThat(s1).isSameAs(s2);

    мҠӨн”„л§Ғ мӢұкёҖнҶӨмқҳ мһҘм җ

    мһҘм җ м„ӨлӘ…
    мһҗлҸҷ кҙҖлҰ¬ private мғқм„ұмһҗ л“ұ л¶Җк°Җ мҪ”л“ң л¶Ҳн•„мҡ”
    DIP мӨҖмҲҳ нҒҙлқјмқҙм–ёнҠёк°Җ мқён„°нҺҳмқҙмҠӨм—җл§Ң мқҳмЎҙ
    OCP мӨҖмҲҳ мҠӨн”„л§Ғмқҙ мқҳмЎҙкҙҖкі„ мЈјмһ…
    н…ҢмҠӨнҠё мҡ©мқҙ Mock к°қмІҙ мЈјмһ… к°ҖлҠҘ
    мң м—°м„ұ мғҒмҶҚ, лӢӨнҳ•м„ұ м§Җмӣҗ

    мӢұкёҖнҶӨ л°©мӢқмқҳ мЈјмқҳмӮ¬н•ӯ: л¬ҙмғҒнғң м„Өкі„

    н•өмӢ¬ мӣҗм№ҷ

    мӢұкёҖнҶӨ к°қмІҙлҠ” мғҒнғңлҘј к°Җм§Җл©ҙ м•Ҳ лҗңлӢӨ (Stateless).

    м—¬лҹ¬ мҠӨл Ҳл“ңк°Җ н•ҳлӮҳмқҳ к°қмІҙлҘј кіөмң н•ҳлҜҖлЎң, мғҒнғң(н•„л“ң)к°Җ мһҲмңјл©ҙ кіөмң  н•„л“ң л¬ём ңк°Җ л°ңмғқн•ңлӢӨ.

    л¬ём ң мҳҲмӢң

    // мң„н—ҳн•ң мҪ”л“ң: кіөмң  н•„л“ң(мғҒнғң) мһҲмқҢ
public class StatefulService {
    private int price; // кіөмң  н•„л“ң вҶ’ мң„н—ҳ!!!

    public void order(String name, int price) {
        System.out.println("name=" + name + " price=" + price);
        this.price = price; // мғҒнғң ліҖкІҪ!
    }

    public int getPrice() {
        return price;
    }
}

    // мҠӨл Ҳл“ң A: к№ҖмІ мҲҳк°Җ 10000мӣҗ мЈјл¬ё
// мҠӨл Ҳл“ң B: к№ҖмҳҒнқ¬к°Җ 20000мӣҗ мЈјл¬ё (кұ°мқҳ лҸҷмӢң)
StatefulService service = ... // мӢұкёҖнҶӨ

// мҠӨл Ҳл“ң A
service.order("к№ҖмІ мҲҳ", 10000);

// мҠӨл Ҳл“ң B (мҠӨл Ҳл“ң A мӮ¬мқҙм—җ лҒјм–ҙл“Ұ)
service.order("к№ҖмҳҒнқ¬", 20000);

// мҠӨл Ҳл“ң Aк°Җ к°ҖкІ© мЎ°нҡҢ
int price = service.getPrice();
// кё°лҢҖ: 10000, мӢӨм ң: 20000 вҶҗ лІ„к·ё!

    мҳ¬л°”лҘё л¬ҙмғҒнғң м„Өкі„

    // м•Ҳм „н•ң мҪ”л“ң: мғҒнғңлҘј м§Җм—ӯ ліҖмҲҳлӮҳ л°ҳнҷҳк°’мңјлЎң мІҳлҰ¬
public class StatelessService {
    // кіөмң  н•„л“ң м—ҶмқҢ!

    public int order(String name, int price) {
        System.out.println("name=" + name + " price=" + price);
        return price; // л°ҳнҷҳк°’мңјлЎң мІҳлҰ¬ (м§Җм—ӯ ліҖмҲҳ)
    }
}

    // ThreadLocalмқ„ мӮ¬мҡ©н•ҙм•ј н•ҳлҠ” кІҪмҡ° (мҠӨл Ҳл“ңл§ҲлӢӨ лҸ…лҰҪм Ғмқё мғҒнғңк°Җ н•„мҡ”)
public class UserContext {
    private static final ThreadLocal<String> currentUser = new ThreadLocal<>();

    public static void setCurrentUser(String user) { currentUser.set(user); }
    public static String getCurrentUser() { return currentUser.get(); }
    public static void clear() { currentUser.remove(); } // мҡ”мІӯ мІҳлҰ¬ нӣ„ л°ҳл“ңмӢң м ңкұ°!
}

    л¬ҙмғҒнғң м„Өкі„ мІҙнҒ¬лҰ¬мҠӨнҠё

    • мқёмҠӨн„ҙмҠӨ ліҖмҲҳм—җ к°’мқ„ м“°лҠ”к°Җ? вҶ’ мң„н—ҳ
    • н•ҙлӢ№ н•„л“ңк°Җ нҒҙлқјмқҙм–ёнҠёл§ҲлӢӨ лӢӨлҘё к°’мқҙм–ҙм•ј н•ҳлҠ”к°Җ? вҶ’ мң„н—ҳ
    • мқҪкё° м „мҡ©мңјлЎңл§Ң мӮ¬мҡ©н•ҳлҠ”к°Җ? вҶ’ м•Ҳм „
    • м§Җм—ӯ ліҖмҲҳ, нҢҢлқјлҜён„°, л°ҳнҷҳк°’мқ„ нҷңмҡ©н•ҳлҠ”к°Җ? вҶ’ м•Ҳм „

    @Configurationкіј CGLIB

    л¬ём ң: мһҗл°” мҪ”л“ң к·ёлҢҖлЎңл©ҙ мӢұкёҖнҶӨмқҙ к№Ём§җ

    @Configuration
public class AppConfig {
    @Bean
    public MemberService memberService() {
        return new MemberServiceImpl(memberRepository()); // memberRepository() нҳём¶ң
    }

    @Bean
    public OrderService orderService() {
        return new OrderServiceImpl(memberRepository(), discountPolicy()); // memberRepository() нҳём¶ң
    }

    @Bean
    public MemberRepository memberRepository() {
        return new MemoryMemberRepository(); // new!
    }
}

    лӢЁмҲңнһҲ мһҗл°” мҪ”л“ңлқјл©ҙ memberRepository()к°Җ 2лІҲ нҳём¶ңлҗҳм–ҙ MemoryMemberRepositoryк°Җ 2к°ң мғқм„ұлҗҳм–ҙм•ј н•  кІғ к°ҷлӢӨ. н•ҳм§Җл§Ң мӢӨм ңлЎңлҠ” 1к°ңл§Ң мғқм„ұлҗңлӢӨ.

    @Configurationмқҳ CGLIB л°”мқҙнҠёмҪ”л“ң мЎ°мһ‘

    мҠӨн”„л§ҒмқҖ @Configurationмқҙ л¶ҷмқҖ нҒҙлһҳмҠӨлҘј л°”мқҙнҠёмҪ”л“ң мЎ°мһ‘ лқјмқҙлёҢлҹ¬лҰ¬(CGLIB)лҘј мӮ¬мҡ©н•ҙ н”„лЎқмӢң нҒҙлһҳмҠӨлЎң көҗмІҙн•ңлӢӨ.
